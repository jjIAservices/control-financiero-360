function renderDashboard(data) { let totals = { total: 0, intereses: 0, amortizado: 0, inversion: 0 }; let catStats = { Hogar: 0, Coche: 0, Ocio: 0, Deudas: 0, Resto: 0 }; ['hogar', 'coche', 'ocio', 'deudas', 'resto'].forEach(id => { const el = document.getElementById(`tbody-${id}`); if(el) el.innerHTML = ''; }); // Procesamos los datos data.forEach(row => { // Buscamos cualquier columna que se parezca a "Fecha" o "Importe" const colFecha = row.Fecha || row.fecha || row[Object.keys(row)[0]]; const colImporte = row.Importe || row.importe || row.monto || row.Monto; if (!colFecha || colFecha.trim() === "" || !colImporte) return; // Limpiar el número de símbolos extraños const monto = parseFloat(colImporte.toString().replace(/[$.€ ]/g, '').replace('.','').replace(',','.')) || 0; totals.total += monto; // Identificar categoría let cat = (row.Categoría || row.categoria || 'Resto').trim(); // Normalizar la primera letra a Mayúscula cat = cat.charAt(0).toUpperCase() + cat.slice(1).toLowerCase(); if(catStats[cat] !== undefined) catStats[cat] += monto; else catStats['Resto'] += monto; const table = document.getElementById(`tbody-${cat.toLowerCase()}`); if(table) { table.innerHTML += ` ${colFecha} ${row.Concepto || row.concepto || 'Sin concepto'} ${monto.toFixed(2)}€ `; } }); // Actualizar textos document.getElementById('gastoGlobal').innerText = totals.total.toLocaleString() + " €"; document.getElementById('status').innerText = 'Sincronizado'; // Dibujar gráfica actualizarGrafica(catStats); }
